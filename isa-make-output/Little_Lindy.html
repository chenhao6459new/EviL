"op |"  (infixr "∨" 30) and
Not  ("¬ _" [40] 40)

text
{* We first define \emph{pseudo-negation}, which
   is essential to the finite Lindenbaum construction. *}

definition (in ClassAx) pneg :: "'a => 'a" ("∼ _" [40] 40) where
"(∼ φ) = (if (∃ ψ. (¬ ψ) = φ) then (SOME ψ. (¬ ψ) = φ) else ¬ φ)"

text
{* We now turn to proving \emph{tertium non datur} for pseudo
   negation, as well as logical equivalence with negation. *}

lemma (in ClassAx) pneg_tnd: "\<turnstile> ∼ φ ∨ φ"
proof cases
  assume "∃ ψ. (¬ ψ) = φ"
  -- "For clarification, the @{thm [source] someI_ex} states: @{thm someI_ex}"
  with someI_ex [where P="% ψ . (¬ ψ) = φ"] 
       pneg_def [where φ="φ"]
     have "(¬ ∼ φ) = φ" by fastsimp
  moreover from dblneg1 have "\<turnstile> ¬ ∼ φ ∨ ∼ φ" .
  with disj_comm mp have "\<turnstile> ∼ φ ∨ ¬ ∼ φ" by blast
  ultimately show ?thesis by simp

  next assume "~ (∃ψ. (¬ ψ) = φ)"
  with pneg_def [where φ="φ"]
     have "(∼ φ) = (¬ φ)" by fastsimp
  moreover from dblneg1 have "\<turnstile> ¬ φ ∨ φ" .
  ultimately show ?thesis by simp 
qed

lemma (in ClassAx) pneg_negimpI: "\<turnstile> ¬ φ -> ∼ φ"
  by (blast intro: pneg_tnd disj_comm mp)

lemma (in ClassAx) pneg_negimpII: "\<turnstile> ∼ φ -> ¬ φ"
proof cases
  assume a: "∃ψ. (¬ ψ) = φ"
  then have "(∼ φ) = (SOME ψ. (¬ ψ) = φ)"
    by (simp add: pneg_def)
  with a 
       someI_ex [where P="% ψ. (¬ ψ) = φ"] 
  have "(¬ (∼ φ)) = φ" by simp
  with dblneg1 have "\<turnstile> ¬ ¬ φ -> ¬ ∼ φ" 
      by simp
  with ax3 [where φ="¬ φ"
              and ψ="∼ φ"] 
  show ?thesis by (blast intro: mp)
  next assume b: "~ (∃ψ. (¬ ψ) = φ)"
  then have "(∼ φ) = (¬ φ)"
       by (simp add: pneg_def)
  with  refl show ?thesis
    by simp
qed

text
{* The following lemma is critical to the
   consistency proof of the Lindenbaum construction. *}

lemma (in ClassAx) cnst:
  assumes ♥: "~ (Γ :\<turnstile> ψ)"
  shows "~ (φ # Γ :\<turnstile> ψ) ∨ ~ ((∼ φ) # Γ :\<turnstile> ψ)"
  using ♥
proof -
  { assume a: "φ # Γ :\<turnstile> ψ" 
       and b: "(∼ φ) # Γ :\<turnstile> ψ"
    from a undisch have "Γ :\<turnstile> φ -> ψ" by simp
   
    moreover from b undisch have "Γ :\<turnstile> ∼ φ -> ψ" 
      by simp
   
    moreover from pneg_tnd have "\<turnstile> (∼ φ) ∨ φ" .
    with lift have "Γ :\<turnstile> (∼ φ) ∨ φ" by fast
   
    moreover note cdil [where Γ="Γ"]
   
    ultimately have "Γ :\<turnstile> ψ" by blast }
  with ♥ show ?thesis by fastsimp
qed

text
{* We now turn to giving a general, finitistic Lindenbaum construction.
   The basis for our method is the following observation: finite sets
   always correspond to some list.  Wielding the axiom of choice, we choose
   a suitable representative list.  We then define a primitive recursive
   function, named {} with type @{typ "'a => 'a list => 'a list => 'a list"}, which 
   first takes a formula @{term [show_types] "ψ::'a"}. It then
   takes a @{term [show_types] "φ::'a"} off the top of the second argument 
   @{term [show_types] "Φ::'a list"} and adds it to the consistently first 
   argument @{term [show_types] "Γ:: 'a list"} if it may be consistently added 
   without proving @{term "ψ"}, and adds @{term "∼ φ"} otherwise.  
   The procedure then recurses. *}

primrec (in ClassAx) lind :: "'a => 'a list => 'a list => 'a list" where
   "lind ψ Γ [] = Γ"
 | "lind ψ Γ (φ # Φ) = (let φ' = if ~(φ # Γ :\<turnstile> ψ) 
                                     then φ
                                     else ∼ φ 
                         in (lind ψ (φ' # Γ) Φ))" 

text
{* We will show the two crucial properties of this construction: 
   (1) either @{term "φ"} or @{term "∼ φ"} are present in the final list 
   for all @{term "φ ∈ Φ"} and (2) if @{term "Γ"} is does not prove @{term "ψ"}, 
   then the resulting construction does not prove @{term "ψ"}.

   We start by proving several basic lemmas, which help us understand
   the results of a lindenbaum construction. As usual, we frequently
   use universal quantification in the statement of lemmas
   to strengthen inductive hypotheses. *}

lemma (in ClassAx) lind_is_mono:
"∀Γ. set Γ ⊆ set (lind ψ Γ Φ)"
proof (induct Φ, simp)
  -- "The base case is trivial, so we only show the inductive step"
  fix φ :: "'a"; fix Φ :: "'a list"
  assume indh: "∀Γ. set Γ ⊆ set (lind ψ Γ Φ)"
  -- "From this assumption, we will show the consequent,"
  --{*where @{term "Γ"} is arbitrary *}
  { fix Γ :: "'a list"
    let ?if = "if ~(φ # Γ :\<turnstile> ψ) 
                  then φ
                  else (∼ φ)"
    -- "Next, observe these two facts:"
    have a: "set Γ ⊆ set (?if # Γ)"
      by fastsimp
    have "lind ψ Γ (φ # Φ) = lind ψ (?if # Γ) Φ" by simp
    with indh have b: "set (?if # Γ) ⊆ set (lind ψ Γ (φ # Φ))"
      by fastsimp
    --{*With these two facts, we can show, for any @{term "x"},*}
    --{*that @{term "x ∈ set (lind ψ Γ (φ # Φ))"} *}
    { fix x :: "'a"; assume "x ∈ set Γ"
      with a have "x ∈ set (?if # Γ)" by fast
      with b have "x ∈ set (lind ψ Γ (φ # Φ))" by fast }
    hence "set Γ ⊆ set (lind ψ Γ (φ # Φ))" by fast }
  -- "This suffices to prove the theorem"
  thus "∀Γ. set Γ ⊆ set (lind ψ Γ (φ # Φ))" by fast
qed

lemma (in ClassAx) lind_is_max:
"∀Γ. ∀φ ∈ set Φ. φ ∈ set (lind ψ Γ Φ) ∨ (∼ φ) ∈ set (lind ψ Γ Φ)"
proof (induct Φ, simp)
  -- "We shall only prove the inductive step"
  fix χ :: "'a"; fix Φ :: "'a list"
  assume ind_hyp: "∀ Γ. ∀φ∈set Φ. φ ∈ set (lind ψ Γ Φ) 
                                   ∨ (∼ φ) ∈ set (lind ψ Γ Φ)"
  -- {* First, let @{term "Γ"} and @{term "φ"} be arbitrary *}
  { fix Γ :: "'a list"; fix φ :: "'a"
    -- {* Next, let we'll use our previous abbreviation *}
    let ?if = "if ~(χ # Γ :\<turnstile> ψ) 
                   then χ
                   else (∼ χ)"
    --{*The following identity will turn out to be crucial *}
    have ♥: 
       "lind ψ (?if # Γ) Φ = lind ψ Γ (χ # Φ)" by fastsimp  
    --{*Next, assume the proper domain conditions for @{term "φ"} *}
    assume ♦: "φ ∈ set (χ # Φ)"
    have "φ ∈ set (lind ψ Γ (χ # Φ)) 
          ∨ (∼ φ) ∈ set (lind ψ Γ (χ # Φ))"
    proof cases
       assume "φ ∈ set Φ"
       with ind_hyp have 
        "φ ∈ set (lind ψ (?if # Γ) Φ) 
         ∨ (∼ φ) ∈ set (lind ψ (?if # Γ) Φ)" by fast
       with ♥ show ?thesis by fastsimp
      next 
       assume "φ ∉ set Φ"
       with ♦ have "φ = χ" by (induct Φ, fastsimp+)
       hence "φ = ?if ∨ (∼ φ) = ?if" by fastsimp
       moreover have "?if ∈ set (?if # Γ)" by fastsimp
       with ♥ lind_is_mono have
        "?if ∈ set (lind ψ Γ (χ # Φ))" by fastsimp
       ultimately show ?thesis by fastsimp
    qed }
  thus 
  "∀Γ. ∀φ∈set (χ # Φ). φ ∈ set (lind ψ Γ (χ # Φ)) 
                        ∨ (∼ φ) ∈ set (lind ψ Γ (χ # Φ))"
   by fast
qed

lemma (in ClassAx) lind_is_bounded:   
  assumes pneg_closed: "(∀ φ ∈ set Φ. (∼ φ) ∈ set Φ)"
    shows "∀ Γ. set (lind ψ Γ Φ) ⊆ set Γ ∪ set Φ"
using pneg_closed
proof -
    -- "We cannot see how to perform this proof through direct"
    -- "induction, so we shall prove it a little, more obliquely." 
    -- "Observe the following statement:"
    let ?pnegset = "% Φ . {φ. ∃ ψ . φ = (∼ ψ) ∧ ψ ∈ set Φ}" 
    from pneg_closed have "?pnegset Φ ⊆ set Φ" by fastsimp
    -- "This inspires an inductive proof which may be performed"
  moreover
    have "∀ Γ. set (lind ψ Γ Φ) ⊆ set Γ ∪ set Φ ∪ ?pnegset Φ"
      proof (induct Φ, simp)
        fix χ :: "'a"; fix Φ :: "'a list"
        assume ind_hyp: 
          "∀Γ. set (lind ψ Γ Φ) ⊆ set Γ ∪ set Φ ∪ ?pnegset Φ"
        --{*As usual, we will show for @{term "Γ"} arbitrary*}
        { fix Γ :: "'a list"
          from ind_hyp have
          "set (lind ψ Γ (χ # Φ)) ⊆ 
                set Γ ∪ set (χ # Φ) ∪ ?pnegset (χ # Φ)"
            by fastsimp }

        thus 
         "∀Γ. set (lind ψ Γ (χ # Φ)) ⊆ 
               set Γ ∪ set (χ # Φ) ∪ ?pnegset (χ # Φ)"
           by fast
      qed
    ultimately show ?thesis by blast
qed

text
{* We now turn to perhaps the key lemma regarding
   Lindenbaum constructions: they preserve consistency! *}

lemma (in ClassAx) lind_is_cnst:
  "∀Γ. ~ (Γ :\<turnstile> ψ) --> ~ (lind ψ Γ Φ :\<turnstile> ψ)"
proof (induct Φ, simp)
   -- "As expected, all we prove is the inductive step."
   fix ψ χ :: "'a"; fix Φ :: "'a list"
   assume ind_hyp: "∀Γ. ~ (Γ :\<turnstile> ψ) --> ~ (lind ψ Γ Φ :\<turnstile> ψ)"
   --{* We shall show the statement of the theorem where @{term "Γ"} is free *}
   { fix Γ :: "'a list"
     let ?if = "if ~(χ # Γ :\<turnstile> ψ) 
                   then χ
                   else (∼ χ)"
     -- "We will need this key fact"
     have key: "lind ψ Γ (χ # Φ) = lind ψ (?if # Γ) Φ" by simp
     -- {* From this, we turn to completing the proof *}
     assume ♥: "~ (Γ :\<turnstile> ψ)" 
     have "~ (lind ψ Γ (χ # Φ) :\<turnstile> ψ)"
     proof cases
       assume a: "χ # Γ :\<turnstile> ψ"
       with ♥ cnst have 
         "~ ((∼ χ) # Γ :\<turnstile> ψ)" by fastsimp
       with ind_hyp have
         "~ (lind ψ ((∼ χ) # Γ) Φ :\<turnstile> ψ)" by blast
       with a key show ?thesis by fastsimp
      next 
       assume b: "~(χ # Γ :\<turnstile> ψ)"
       with ind_hyp have 
         "~ (lind ψ (χ # Γ) Φ :\<turnstile> ψ)" by blast
       with b key show ?thesis by fastsimp
      qed }
   thus " ∀Γ. ~ (Γ :\<turnstile> ψ) --> ~ (lind ψ Γ (χ # Φ) :\<turnstile> ψ)" by fast
qed

text
{* We now give a predicate for atoms, which are maximally 
   consistent sets relative to a finite set @{term "Φ"}.  We 
   shall prove that they contain a formula @{term "φ ∈ Φ"}
   if and only if they deduce that formula. While we are at it,
   we shall prove that in the same context, 
   @{term "(φ ∉ Γ) = (∼ φ ∈ Γ)"} *}

definition (in ClassAx) 
Atoms :: "'a set => 'a set => bool" ("At") where
"At Φ Γ ≡   Γ ⊆ Φ
            ∧ (∀φ ∈ Φ. φ ∈ Γ ∨ (∼ φ) ∈ Γ)
            ∧ ~(list Γ :\<turnstile> ⊥)"

lemma (in ClassAx) coincidence:
assumes A: "finite Φ"
    and B: "Γ ∈ At(Φ)"
    and C: "φ ∈ Φ"
    and D: "P (∼ φ) = (~ P φ)"
shows "(φ ∈ Γ) = (list Γ :\<turnstile> φ)"
  and "P φ = (list Γ :\<turnstile> φ) ==> P (∼ φ) = (list Γ :\<turnstile> ∼ φ)"
using A B C D
proof -
  --"We shall first make some observations:"
  from A B C
       mem_def [where x="Γ"
                  and S="At(Φ)"]
       Atoms_def [where Γ="Γ" 
                    and Φ="Φ"]
  have "Γ ⊆ Φ" 
   and E: "φ ∈ Γ ∨ (∼ φ) ∈ Γ"
   and F: "~(list Γ :\<turnstile> ⊥)"
         by fastsimp+
  with A finite_subset [where A="Γ"
                          and B="Φ"]
       set_list [where A="Γ"]
  have G: "Γ = set (list Γ)" by fastsimp
 
 --"Our coincidence lemma has two statements; here is the first:"
  show H: "(φ ∈ Γ) = (list Γ :\<turnstile> φ)"
  proof -
    --"The first direction in this case is trivial"
    from G lift_elm have "φ ∈ Γ ==> list Γ :\<turnstile> φ"
       by blast
    --"The other direction is evidently more challenging"
    moreover
    { assume ♥: "list Γ :\<turnstile> φ"
      have "φ ∈ Γ"
      proof -
        --"The proof proceeds by contradiction:"
        { assume "φ ∉ Γ"
          with E have "(∼ φ) ∈ Γ" by fastsimp
          with G lift_elm have "list Γ :\<turnstile> ∼ φ" by blast
          with pneg_negimpII
               lift [where Γ="list Γ"]
               lift_mp [where Γ="list Γ"]
          have "list Γ :\<turnstile> ¬ φ" by blast
          with ♥ lift_mp have "list Γ :\<turnstile> ⊥" by fast
          with F have "False" by fast }
        thus ?thesis by fast
      qed }
    ultimately show ?thesis by blast
  qed

  --"We now turn to the second statement; but we shall first"
  --"make a critical observation:"
  have I: "φ ∉ Γ = (list Γ :\<turnstile> ∼ φ)"
  proof -
  --"Left to right:"
    { assume "φ ∉ Γ"
        with E have "(∼ φ) ∈ Γ" by auto
        with G lift_elm
              have "list Γ :\<turnstile> ∼ φ" by blast }
  moreover
  --"Right to left:"
    { assume "φ ∈ Γ"
         and "list Γ :\<turnstile> ∼ φ"
      moreover with G lift_elm 
               have "list Γ :\<turnstile> φ" by blast
      moreover note F
           pneg_negimpII [where φ="φ"]
           lift [where Γ="list Γ"]
           lift_mp [where Γ="list Γ"]
      ultimately have "False" by blast }
    ultimately show ?thesis by auto
  qed

  --"This is enough to finally show the second statement:"
  assume "P φ = (list Γ :\<turnstile> φ)"
  with D H have "P (∼ φ) = (φ ∉ Γ)" by simp
  with I show "P (∼ φ) = (list Γ :\<turnstile> (∼ φ))" by simp
qed

text
{* We finally turn to presenting the finitary Lindenbaum Lemma. 
   It is in terms of 
   atoms that we shall phrase the primary result we have been leading up to: *}

lemma (in ClassAx) little_lindy:
  assumes A: "finite Φ"
      and B: "∀φ ∈ Φ. (∼ φ) ∈ Φ"
      and C: "Γ ⊆ Φ"
      and D: "~(list Γ :\<turnstile> ψ)" 
    shows "∃ Γ'. At Φ Γ'
                 ∧ Γ ⊆ Γ'
                 ∧ ~(list Γ' :\<turnstile> ψ)"
using A B C D
proof -
  from A C finite_subset have 
     "finite Γ" by fastsimp
  with set_list [where A="Γ"] have
     E: "Γ = set (list Γ)" by auto
  from A set_list [where A="Φ"] have
     F: "Φ = set (list Φ)" by auto
  let ?lindy = "set (lind ψ (list Γ) (list Φ))" 
  from set_of_list_is_finite have
     "finite ?lindy" by fastsimp
  with set_list [where A="?lindy"] have
     G: "?lindy = set (list ?lindy)" by auto

  -- {* We have many things to prove: *}
  from B C E F 
       lind_is_bounded [where Φ="list Φ"]
  have I: "?lindy ⊆ Φ" 
    by blast

  from F lind_is_max 
  have II: "∀φ ∈ Φ. φ ∈ ?lindy ∨ (∼ φ) ∈ ?lindy"
    by fastsimp
  
  from D G
       lind_is_cnst [where Φ="list Φ"]
       lift_eq [where Γ="lind ψ (list Γ) (list Φ)" 
                  and Ψ="list ?lindy"]
  have III: "~ (list ?lindy :\<turnstile> ψ)" 
    by blast

  from III
       expls [where φ="ψ"]
       lift  [where Γ="list ?lindy"]
       lift_mp [where Γ="list ?lindy"]
  have IV: "~ (list ?lindy :\<turnstile> ⊥)" 
    by blast
  
  from E
       lind_is_mono [where ψ="ψ"
                       and Φ="list Φ"]
  have V: "Γ ⊆ ?lindy"
    by blast

  from I II IV 
       Atoms_def [where Φ="Φ"
                   and Γ="?lindy"]
  have VI: "At Φ ?lindy" by fastsimp

  from III V VI show ?thesis by fastsimp
qed
