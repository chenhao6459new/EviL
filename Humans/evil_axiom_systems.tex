 \vdash \BB_X \phi \IFF \BB_X \BB_X \phi & \vdash \BBI_X \phi \IFF \BBI_X \BBI_X \phi & \vdash \PP_X \IFF \BB_X \PP_X\end{eqnarray*}
%&\vdash \PP_X \IFF \DDI_X \PP_X \\

\end{lemma}
